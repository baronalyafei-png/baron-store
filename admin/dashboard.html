    /* ========================================= BARON CORE ENGINE POS + Inventory + Invoices + Debts + Stats ========================================= */ let inventory = [ { id: 1, name: "سماعة بلوتوث", price: 150, stock: 10, image: "https://via.placeholder.com/100" }, { id: 2, name: "هاتف ذكي", price: 2500, stock: 5, image: "https://via.placeholder.com/100" }, { id: 3, name: "كيبورد ميكانيكي", price: 320, stock: 8, image: "https://via.placeholder.com/100" } ]; let cart = []; let invoices = []; let debts = []; let taxRate = 0.15; let selectedPayment = "cash"; /* ================= PRODUCTS ================= */ function renderProducts(){ const grid=document.getElementById("products-grid-pos"); if(!grid) return; grid.innerHTML=""; inventory.forEach(p=>{ grid.innerHTML+=`
    ${p.name}
    ${p.price} ر.س
    الكمية: ${p.stock}
    `; }); } /* ================= CART ================= */ function addToCart(id){ const product=inventory.find(p=>p.id===id); if(product.stock<=0) return alert("المنتج غير متوفر"); const existing=cart.find(i=>i.id===id); if(existing){ if(existing.qty>=product.stock) return alert("الكمية غير كافية"); existing.qty++; }else{ cart.push({...product, qty:1}); } renderCart(); } function changeQty(id,amount){ const item=cart.find(i=>i.id===id); if(!item) return; item.qty+=amount; if(item.qty<=0){ cart=cart.filter(i=>i.id!==id); } renderCart(); } function renderCart(){ const list=document.getElementById("cart-items-modern"); if(!list) return; list.innerHTML=""; let subtotal=0; cart.forEach(item=>{ let total=item.price*item.qty; subtotal+=total; list.innerHTML+=`
  • ${item.name} × ${item.qty}
    ${total.toFixed(2)} ر.س
    • `; }); let tax=subtotal*taxRate; let total=subtotal+tax; document.getElementById("subtotal").innerText=subtotal.toFixed(2)+" ر.س"; document.getElementById("tax").innerText=tax.toFixed(2)+" ر.س"; document.getElementById("total").innerText=total.toFixed(2)+" ر.س"; calculateChange(); } /* ================= PAYMENT ================= */ document.querySelectorAll(".payment-option").forEach(option=>{ option.addEventListener("click",function(){ document.querySelectorAll(".payment-option").forEach(o=>o.classList.remove("active")); this.classList.add("active"); selectedPayment=this.dataset.payment; document.getElementById("customer-select").style.display = selectedPayment==="credit"?"block":"none"; }); }); function calculateChange(){ let paid=parseFloat(document.getElementById("paid-amount").value)||0; let total=parseFloat(document.getElementById("total").innerText); let change=paid-total; document.getElementById("change-display").innerText = change>0?change.toFixed(2)+" ر.س":"0.00 ر.س"; } document.getElementById("paid-amount")?.addEventListener("input",calculateChange); /* ================= COMPLETE SALE ================= */ document.getElementById("complete-sale-btn")?.addEventListener("click",()=>{ if(cart.length===0) return alert("السلة فارغة"); let subtotal=parseFloat(document.getElementById("subtotal").innerText); let tax=parseFloat(document.getElementById("tax").innerText); let total=parseFloat(document.getElementById("total").innerText); // خصم المخزون cart.forEach(item=>{ const product=inventory.find(p=>p.id===item.id); product.stock-=item.qty; }); let invoiceNumber="INV-"+(invoices.length+1).toString().padStart(4,"0"); let party = selectedPayment==="credit" ? document.getElementById("customer-list").selectedOptions[0].text : "عميل نقدي"; let status = selectedPayment==="credit"?"معلقة":"مدفوعة"; invoices.push({ number:invoiceNumber, party, total, status, date:new Date().toLocaleDateString() }); // إنشاء دين إذا آجل if(selectedPayment==="credit"){ debts.push({ invoice:invoiceNumber, name:party, amount:total, settled:false }); } updateStats(); renderInvoices(); renderDebts(); renderProducts(); alert("تم إنشاء الفاتورة "+invoiceNumber+" ✅"); cart=[]; document.getElementById("paid-amount").value=""; renderCart(); }); /* ================= INVOICES ================= */ function renderInvoices(){ const table=document.getElementById("invoices-table-body"); if(!table) return; table.innerHTML=""; invoices.forEach(inv=>{ table.innerHTML+=` ${inv.number} بيع POS ${inv.party} ${inv.date} ${inv.total.toFixed(2)} ر.س ${inv.status} — `; }); } /* ================= DEBTS ================= */ function renderDebts(){ const grid=document.getElementById("rep-debts-grid"); if(!grid) return; grid.innerHTML=""; debts.forEach((d,index)=>{ if(d.settled) return; grid.innerHTML+=`
    ${d.name}
    فاتورة: ${d.invoice}
    المبلغ: ${d.amount.toFixed(2)} ر.س
    `; }); } function settleDebt(index){ debts[index].settled=true; let inv=invoices.find(i=>i.number===debts[index].invoice); if(inv) inv.status="مدفوعة"; updateStats(); renderInvoices(); renderDebts(); } /* ================= STATS ================= */ function updateStats(){ let totalDebts = debts .filter(d=>!d.settled) .reduce((sum,d)=>sum+d.amount,0); document.getElementById("total-debts").innerText= totalDebts.toFixed(2)+" ر.س"; document.getElementById("total-rep-debts").innerText= totalDebts.toFixed(2)+" ر.س"; document.getElementById("total-supplier-debts").innerText="0 ر.س"; } /* ================= INIT ================= */ document.addEventListener("DOMContentLoaded",()=>{ renderProducts(); renderInvoices(); renderDebts(); updateStats(); });
